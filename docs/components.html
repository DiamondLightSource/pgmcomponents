205    def compute_angles(self):
206        """
207        Compute the incident and diffraction angles of the grating.
208        The incident angle is calculated from the diffraction angle
209        using the fixed focus constant.
210
211        Returns
212        -------
213        alpha : float
214            The incident angle in degrees
215
216        beta : float
217            The diffraction angle in degrees
218        
219        """
220        wavelength = wavelength(self.energy)
221        lambda_u = self.order*self.line_density*1000*wavelength/(1-self.cff**2)
222        sin_alpha = lambda_u + np.sqrt(1+lambda_u**2*self.cff**2)
223        self._alpha = np.rad2deg(np.arcsin(sin_alpha))
224        self._beta = -np.rad2deg(np.arccos(np.cos(np.arcsin(sin_alpha))*self.cff))
225
226        return self._alpha, self._beta

298    def compute_corners(self):
299        beta_g = np.deg2rad(self._beta + 90)
300        l = self._length()
301        w = self._width()
302        d = self._height()        
303        #Bottom left back
304        blbz = -(l/2)*np.cos(beta_g)
305        blby = -(l/2)*np.sin(beta_g)
306        blbx = -w/2
307        blb = Point3D(blbx, blby, blbz)
308        #Bottom right back
309        brbz = blbz
310        brby = blby
311        brbx = w/2
312        brb = Point3D(brbx, brby, brbz)
313
314        #Bottom left front
315        blfz = (l/2)*np.cos(beta_g)
316        blfy = (l/2)*np.sin(beta_g)
317        blfx = -w/2
318        blf = Point3D(blfx, blfy, blfz)
319
320        #Bottom right front
321        brfz = blfz
322        brfy = blfy
323        brfx = w/2
324        brf = Point3D(brfx, brfy, brfz)
325
326        #Top left back
327        tlbz = blbz - d*np.sin(beta_g)
328        tlby = blby + d*np.cos(beta_g)
329        tlbx = -w/2
330        tlb = Point3D(tlbx, tlby, tlbz)
331
332        #Top right back
333        trbz = brbz - d*np.sin(beta_g)
334        trby = brby + d*np.cos(beta_g)
335        trbx = w/2
336        trb = Point3D(trbx, trby, trbz)
337
338        #Top left front
339        tlfz = blfz - d*np.sin(beta_g)
340        tlfy = blfy + d*np.cos(beta_g)
341        tlfx = -w/2
342        tlf = Point3D(tlfx, tlfy, tlfz)
343
344        #Top right front
345        trfz = brfz - d*np.sin(beta_g)
346        trfy = brfy + d*np.cos(beta_g)
347        trfx = w/2
348        trf = Point3D(trfx, trfy, trfz)
349
350        self._grating_plane = Plane(
351            Point3D(blfx, blfy, blfz),
352            Point3D(brfx, brfy, brfz),
353            Point3D(blbx, blby, blbz)
354        )
355
356        self._corners = np.array([
357            blb,
358            brb,
359            blf,
360            brf,
361            tlb,
362            trb,
363            tlf,
364            trf
365        ])
366
367        return self._corners

369    def reflect(self, *args):
370        """
371        A method to reflect rays off the grating.
372
373        Parameters
374        ----------
375        *args : Ray3D
376            The rays to be reflected
377
378        Returns
379        -------
380        reflected_rays : list
381            A list of reflected rays
382
383        """
384        reflected_rays = []
385        
386        if len(args) == 0:
387            raise ValueError("Expected at least one ray")
388        
389        if type(args[0]) == list:
390            args = args[0]
391        
392
393        for index, ray in enumerate(args):
394            if not isinstance(ray, Ray3D):
395                raise TypeError("Expected Ray3D object")
396            try:
397                plane_intersection = self._grating_plane.intersectQ(ray)
398            except ValueError:
399                print(f'Ray of index {index} does not intersect grating, tread with caution!')
400                continue
401            ray_array = ray.vector
402            grating_normal = self._grating_plane.normal
403            reflected_ray_array = ray_array - 2 * np.dot(ray_array, grating_normal) * grating_normal
404            reflected_ray_array = reflected_ray_array / np.linalg.norm(reflected_ray_array)
405            reflected_ray = Ray3D(plane_intersection, reflected_ray_array)
406            reflected_rays.append(reflected_ray)
407        
408        return reflected_rays
