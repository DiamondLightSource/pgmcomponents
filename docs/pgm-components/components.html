 252    def compute_angles(self):
 253        """
 254        Compute the incident and diffraction angles of the grating.
 255        The incident angle is calculated from the diffraction angle
 256        using the fixed focus constant.
 257
 258        Returns
 259        -------
 260        alpha : float
 261            The incident angle in degrees
 262
 263        beta : float
 264            The diffraction angle in degrees
 265        
 266        """
 267        wavelength = self.wavelength(self.energy)
 268        lambda_u = self.order*self.line_density*1000*wavelength/(1-self.cff**2)
 269        sin_alpha = lambda_u + np.sqrt(1+lambda_u**2*self.cff**2)
 270        self._alpha = np.rad2deg(np.arcsin(sin_alpha))
 271        self._beta = -np.rad2deg(np.arccos(np.cos(np.arcsin(sin_alpha))*self.cff))
 272
 273        return self._alpha, self._beta

 275    def diffract(self, *args):
 276        """
 277        A method to diffract rays off the grating.
 278
 279        Parameters
 280        ----------
 281        rays : list
 282            The rays to be diffracted
 283
 284        Returns
 285        -------
 286        diffracted_rays : list
 287            A list of diffracted rays
 288
 289        """
 290        diffracted_rays = []
 291        
 292        if len(args) == 0:
 293            raise ValueError("Expected at least one ray")
 294        
 295        if type(args[0]) == list:
 296            args = args[0]
 297
 298
 299        for index, ray in enumerate(args):
 300            if not isinstance(ray, Ray3D):
 301                raise TypeError("Expected Ray3D object")
 302            raydotplane = ray.vector.dot(self._grating_plane.normal)
 303            angle = np.arccos(raydotplane/self._grating_plane.normal.norm())
 304            alpha = np.rad2deg(np.pi/2-angle)
 305            beta = self.calc_beta(alpha, self._line_density, self._energy, self._order)
 306            diff_ray = self.reflect(ray)
 307            angle = -90 - beta - alpha
 308            diff_ray.vector[2] += np.cos(np.deg2rad(angle))
 309            diff_ray.vector[1] += np.sin(np.deg2rad(angle))
 310            diff_ray.vector.normalize()
 311            diffracted_rays.append(diff_ray)
 312        return diffracted_rays

 351    def compute_corners(self):
 352        
 353        beta = np.deg2rad(self._beta)
 354        beta_g = np.deg2rad(self._beta + 90)
 355        l = self._length()
 356        w = self._width()
 357        d = self._height()        
 358        #Bottom left back
 359        blbz = (l/2)*np.sin(beta)
 360        blby = -(l/2)*np.cos(beta)
 361        blbx = -w/2
 362        blb = Point3D(blbx, blby, blbz)
 363        #Bottom right back
 364        brbz = blbz
 365        brby = blby
 366        brbx = w/2
 367        brb = Point3D(brbx, brby, brbz)
 368
 369        #Bottom left front
 370        blfz = -(l/2)*np.sin(beta)
 371        blfy = (l/2)*np.cos(beta)
 372        blfx = -w/2
 373        blf = Point3D(blfx, blfy, blfz)
 374
 375        #Bottom right front
 376        brfz = blfz
 377        brfy = blfy
 378        brfx = w/2
 379        brf = Point3D(brfx, brfy, brfz)
 380
 381        #Top left back
 382        tlbz = blbz - d*np.cos(beta)
 383        tlby = blby - d*np.sin(beta)
 384        tlbx = -w/2
 385        tlb = Point3D(tlbx, tlby, tlbz)
 386
 387        #Top right back
 388        trbz = brbz - d*np.cos(beta)
 389        trby = brby - d*np.sin(beta)
 390        trbx = w/2
 391        trb = Point3D(trbx, trby, trbz)
 392
 393        #Top left front
 394        tlfz = blfz - d*np.cos(beta)
 395        tlfy = blfy - d*np.sin(beta)
 396        tlfx = -w/2
 397        tlf = Point3D(tlfx, tlfy, tlfz)
 398
 399        #Top right front
 400        trfz = brfz - d*np.cos(beta)
 401        trfy = brfy - d*np.sin(beta)
 402        trfx = w/2
 403        trf = Point3D(trfx, trfy, trfz)
 404
 405        self._grating_plane = Plane(
 406            Point3D(blfx, blfy, blfz),
 407            Point3D(brfx, brfy, brfz),
 408            Point3D(blbx, blby, blbz)
 409        )
 410
 411        self._corners = np.array([
 412            blb,
 413            brb,
 414            blf,
 415            brf,
 416            tlb,
 417            trb,
 418            tlf,
 419            trf
 420        ])
 421
 422        return self._corners

 424    def reflect(self, *args):
 425        """
 426        A method to reflect rays off the grating.
 427
 428        Parameters
 429        ----------
 430        *args : Ray3D
 431            The rays to be reflected
 432
 433        Returns
 434        -------
 435        reflected_rays : list
 436            A list of reflected rays
 437
 438        """
 439        reflected_rays = []
 440        
 441        if len(args) == 0:
 442            raise ValueError("Expected at least one ray")
 443        
 444        if type(args[0]) == list:
 445            args = args[0]
 446        
 447
 448        for index, ray in enumerate(args):
 449            if not isinstance(ray, Ray3D):
 450                raise TypeError("Expected Ray3D object")
 451            try:
 452                plane_intersection = self._grating_plane.intersectQ(ray)
 453            except ValueError:
 454                print(f'Ray of index {index} does not intersect grating, tread with caution!')
 455                continue
 456            ray_array = ray.vector
 457            grating_normal = self._grating_plane.normal
 458            reflected_ray_array = ray_array - 2 * np.dot(ray_array, grating_normal) * grating_normal
 459            reflected_ray_array = reflected_ray_array / np.linalg.norm(reflected_ray_array)
 460            reflected_ray = Ray3D(plane_intersection, reflected_ray_array)
 461            reflected_rays.append(reflected_ray)
 462        
 463        return reflected_rays

 755    def compute_corners(self):
 756        """
 757        Compute the corners of the mirror in the global coordinate system,
 758        in addition to the plane and normal of the mirror.
 759    
 760
 761        Returns
 762        -------
 763        corners : array_like
 764            The corners of the mirror in the global coordinate system
 765        """
 766        cot = lambda x: 1/np.tan(x)
 767        theta = np.deg2rad(self.theta)
 768        theta_g = 90 - self._theta
 769        theta_g = np.deg2rad(theta_g)
 770        a = self._hoffset
 771        c = self._voffset
 772        v = self._axis_voffset
 773        h = self._axis_hoffset
 774        w = self._width()
 775        l = self._length()
 776        d = self._height()
 777        #Top left front
 778
 779        tlfz = -((a - c * cot(theta)) * np.sin(theta)) + h
 780        tlfy = -(c / np.sin(theta) + 
 781                 (a - c*cot(theta)) * np.cos(theta)) + v
 782        tlfx = -w/2
 783        tlf = Point3D(tlfx, tlfy, tlfz)
 784
 785        #Bottom left front
 786        blfz = tlfz + d*np.cos(theta)
 787        blfy = tlfy - d*np.sin(theta)        
 788        blfx = -w/2
 789        blf = Point3D(blfx, blfy, blfz)
 790
 791        #Top right front
 792        trfz = tlfz
 793        trfy = tlfy
 794        trfx = w/2
 795        trf = Point3D(trfx, trfy, trfz)
 796
 797        #Bottom right front
 798        brfz = blfz
 799        brfy = blfy
 800        brfx = w/2
 801        brf = Point3D(brfx, brfy, brfz)
 802
 803        #Top left back
 804        tlbz = tlfz - l*np.sin(theta)
 805        tlby = tlfy - l*np.cos(theta)
 806        tlbx = -w/2
 807        tlb = Point3D(tlbx, tlby, tlbz)
 808
 809        #Bottom left back
 810        blbz = tlbz + d*np.cos(theta)
 811        blby = tlby - d*np.sin(theta)
 812        blbx = -w/2

 814
 815        #Top right back
 816        trbz = tlbz
 817        trby = tlby
 818        trbx = w/2
 819        trb = Point3D(trbx, trby, trbz)
 820
 821        #Bottom right back
 822        brbz = blbz
 823        brby = blby
 824        brbx = w/2
 825        brb = Point3D(brbx, brby, brbz)

 831            trf,
 832            blf,
 833            brf,
 834            tlb,
 835            trb,

 868        ----------
 869        *args : Ray3D

 873        -------
 874        reflected_rays : list
 875            A list of reflected rays
 876
 877        """
 878        reflected_rays = []
 879        
 880        if len(args) == 0:
 881            raise ValueError("Expected at least one ray")
 882        
 883        if type(args[0]) == list:
 884            args = args[0]
 885        
 886
 887        for index, ray in enumerate(args):
 888            if not isinstance(ray, Ray3D):
 889                raise TypeError("Expected Ray3D object")
 890            try:
 891                plane_intersection = self._plane.intersectQ(ray)
 892            except ValueError:
 893                print(f'Ray of index {index} does not intersect mirror, tread with caution!')
 894                continue
 895            ray_array = ray.vector
 896            mirror_normal = self._plane.normal
 897            reflected_ray_array = ray_array - 2 * np.dot(ray_array, mirror_normal) * mirror_normal
 898            reflected_ray_array = reflected_ray_array / np.linalg.norm(reflected_ray_array)
 899            reflected_ray = Ray3D(plane_intersection, reflected_ray_array)
 900            reflected_rays.append(reflected_ray)
 901

794        trfy = tlfy
795        trfx = w/2

803
804        #Top left back
